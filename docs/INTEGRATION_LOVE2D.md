# LГ–VE2D Integration Guide

This guide shows how to use Lex with LГ–VE2D to generate Lua modules for your game content.

## Quick Start

### 1. Install Lex

```bash
# Windows
irm https://raw.githubusercontent.com/David-Imperium/Lex-compiler/main/install.ps1 | iex

# Linux/macOS
curl -fsSL https://raw.githubusercontent.com/David-Imperium/Lex-compiler/main/install.sh | bash
```

### 2. Create Your Content

Create a file `content/towers.lex`:

```lex
structure Turret {
    era: Ancient
    name: "Basic Turret"
    cost: { Gold: 50 }
    damage: 10
    range: 2
}

structure Cannon {
    era: Medieval
    name: "Cannon"
    cost: { Gold: 100 }
    damage: 25
    range: 3
}
```

### 3. Compile to Lua

```bash
lexc content/towers.lex -t lua -o content_compiled/
```

This generates `content_compiled/towers.lua`:

```lua
-- Auto-generated by Lex Compiler
local LexHelpers = {}

function LexHelpers.has_technology(tech_id, state)
    return state and state.technologies and state.technologies[tech_id]
end

function LexHelpers.has_structure(structure_id, state)
    return state and state.structures and state.structures[structure_id]
end

local Structures = {}
Structures.Turret = {
    id = "Turret",
    type = "structure",
    era = "Ancient",
    name = "Basic Turret",
    cost = {Gold = 50},
    damage = 10,
    range = 2
}
Structures.Cannon = {
    id = "Cannon",
    type = "structure",
    era = "Medieval",
    name = "Cannon",
    cost = {Gold = 100},
    damage = 25,
    range = 3
}

return {
    structures = Structures,
    LexHelpers = LexHelpers
}
```

### 4. Load in LГ–VE2D

```lua
-- main.lua
local towers_content = require("content_compiled.towers")

local towers = towers_content.structures
local LexHelpers = towers_content.LexHelpers

function love.load()
    -- Access tower data
    print(towers.Turret.name)  -- "Basic Turret"
    print(towers.Turret.damage)  -- 10
end

function get_tower(tower_id)
    return towers[tower_id]
end
```

### 5. Use in Your Game

```lua
-- tower.lua
local towers = require("content_compiled.towers").structures

local Tower = {}
Tower.__index = Tower

function Tower.new(tower_id, x, y)
    local data = towers[tower_id]
    if not data then
        error("Unknown tower: " .. tower_id)
    end
    
    local self = setmetatable({}, Tower)
    self.id = tower_id
    self.x = x
    self.y = y
    self.data = data
    self.hp = data.hp or 100
    return self
end

function Tower:get_damage()
    return self.data.damage
end

function Tower:get_range()
    return self.data.range * 64  -- Convert to pixels
end

return Tower
```

## Conditions

Lex generates helper functions for conditions:

```lex
structure SteamFactory {
    era: Steampunk
    cost: { Gold: 100 }
    
    available_if has_technology(SteamEngine) {
    }
}
```

In your game:

```lua
-- game_state.lua
local GameState = {}

GameState.technologies = {}
GameState.structures = {}
GameState.resources = {Gold = 1000}

function GameState.has_technology(tech_id)
    return GameState.technologies[tech_id] ~= nil
end

function GameState.has_structure(structure_id)
    return GameState.structures[structure_id] ~= nil
end

return GameState
```

```lua
-- Check conditions
local content = require("content_compiled.structures")
local LexHelpers = content.LexHelpers
local GameState = require("game_state")

function can_build(structure_id)
    local data = content.structures[structure_id]
    if not data then return false end
    
    -- Check conditions
    if data.conditions then
        for _, cond in ipairs(data.conditions) do
            if cond.trigger == "available_if" then
                local expr = cond.expression
                if expr:match("has_technology%((.+)%)") then
                    local tech_id = expr:match("has_technology%((.+)%)")
                    if not GameState.has_technology(tech_id) then
                        return false
                    end
                end
            end
        end
    end
    
    -- Check cost
    for resource, amount in pairs(data.cost or {}) do
        if (GameState.resources[resource] or 0) < amount then
            return false
        end
    end
    
    return true
end
```

## Hot Reloading

LГ–VE2D can hot reload Lua modules. Combine with Lex watch mode:

Terminal 1:
```bash
lexc content/*.lex -t lua -o content_compiled/ --watch
```

In your game:
```lua
-- Add hot reload key
function love.keypressed(key)
    if key == "r" then
        -- Reload all content
        package.loaded["content_compiled.towers"] = nil
        package.loaded["content_compiled.enemies"] = nil
        
        towers = require("content_compiled.towers").structures
        enemies = require("content_compiled.enemies").units
        
        print("Content reloaded!")
    end
end
```

## Using with Love2D Backend

Lex has a dedicated Love2D backend that generates framework-optimized output:

```bash
lexc content.lex -t love2d -o content_compiled/
```

The Love2D backend includes:
- Module format for `require()`
- LexHelper functions for conditions
- Proper data structures for game use

### 2. Don't Edit Generated Files

Generated files are overwritten. Create wrapper modules instead:

```lua
-- modules/tower_data.lua
local raw_towers = require("content_compiled.towers").structures

local TowerData = {}

-- Add computed properties
function TowerData.get(tower_id)
    local data = raw_towers[tower_id]
    if not data then return nil end
    
    -- Add computed values
    data.range_pixels = data.range * 64
    data.total_cost = 0
    for _, v in pairs(data.cost or {}) do
        data.total_cost = data.total_cost + v
    end
    
    return data
end

return TowerData
```

## Example: Tower Defense Game

```lua
-- main.lua
local Content = require("modules.content_loader")
local GameState = require("modules.game_state")

local towers = {}
local enemies = {}

function love.load()
    GameState.init()
end

function love.update(dt)
    -- Update enemies
    for i = #enemies, 1, -1 do
        local enemy = enemies[i]
        enemy:update(dt)
        
        if enemy:reached_end() then
            GameState.lives = GameState.lives - 1
            table.remove(enemies, i)
        end
    end
    
    -- Update towers (targeting, shooting)
    for _, tower in ipairs(towers) do
        tower:update(dt, enemies)
    end
end

function love.draw()
    -- Draw towers
    for _, tower in ipairs(towers) do
        tower:draw()
    end
    
    -- Draw enemies
    for _, enemy in ipairs(enemies) do
        enemy:draw()
    end
    
    -- Draw UI
    love.graphics.print("Gold: " .. GameState.resources.Gold, 10, 10)
    love.graphics.print("Lives: " .. GameState.lives, 10, 30)
end

function love.mousepressed(x, y, button)
    if button == 1 then
        -- Try to place tower
        local tower_cost = Content.get_tower("Turret").cost.Gold
        if GameState.resources.Gold >= tower_cost then
            local Tower = require("systems.tower")
            table.insert(towers, Tower.new("Turret", x, y))
            GameState.resources.Gold = GameState.resources.Gold - tower_cost
        end
    end
end
```

## Troubleshooting

### "Module not found"

Make sure the output directory is in `package.path`:

```lua
package.path = package.path .. ";content_compiled/?.lua"
```

Or use relative requires:

```lua
local towers = require("content_compiled.towers")
```

### "Conditions not working"

Conditions are stored as strings. Use pattern matching or LexHelpers:

```lua
function evaluate_condition(expr)
    local tech_id = expr:match("has_technology%((.+)%)")
    if tech_id then
        return GameState.has_technology(tech_id)
    end
    return true
end
```

### "Hot reload not working"

Make sure to set the module to `nil` in `package.loaded` before re-requiring:

```lua
package.loaded["content_compiled.towers"] = nil
local towers = require("content_compiled.towers")
```
