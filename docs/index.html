Clean, Declarative Syntax

game.lex 
structure Turret {
    name: "Basic Turret"
    cost: { Gold: 50 }
    damage: 10
    range: 2
}

structure Cannon {
    name: "Heavy Cannon"
    cost: { Gold: 100 }
    damage: 25
    
    available_if has_tech(Gunpowder) {}
}
terminal 
# Compile to all targets
$ lexc game.lex -t lua,json,gd,cs

# Or just what you need
$ lexc game.lex -t gd -o godot/content/

# With AI context
$ lexc game.lex --context md

# Query your data
$ lexc game.lex --query "What costs the most?"
